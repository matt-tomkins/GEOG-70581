3.3 Assigning variables

# Assigns values to variables
bob <- 1
bill <- 2

# Addition
bob + bill
## [1] 3
# Multiplication
bob * bill
## [1] 2
# Division
bob / bill
## [1] 0.5
# Subtraction
bob - bill
## [1] -1

3.4 Data types

# Character
a <- "Bob"

# Numeric
b <- 3.567

# Integer
c <- 3

# Logical
d <- TRUE

3.5 Vectors

# Creating a list of numbers
ls <- c(1, 2, 3, 4, 5)
ls
## [1] 1 2 3 4 5
# Creating a sequence
sq <- seq(1,5,1)
sq
## [1] 1 2 3 4 5
# Creating a repetitive list
rp <- rep(10, 5)
rp
## [1] 10 10 10 10 10
# Multiplying together
ls * rp
## [1] 10 20 30 40 50

We may want to index using df[row.index, column.index]
